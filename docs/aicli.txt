#### 3.2.4 Exec Service（执行引擎）

- **沙箱执行**：修复操作在隔离环境执行（Docker / 临时 Git 工作区）
- **Git 操作**：自动创建分支、提交变更、生成 PR

5.1 Bug 记录
sql
CREATE TABLE bug_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    project_id UUID NOT NULL,
    session_id VARCHAR(64) NOT NULL,
    
    -- 错误签名（用于查重）
    error_signature VARCHAR(255) NOT NULL,  -- 如: "ModuleNotFoundError:utils/helper"
    error_type VARCHAR(50) NOT NULL,         -- 如: "ModuleNotFound", "NullPointer"
    
    -- 原始输入
    error_input TEXT NOT NULL,               -- 用户粘贴的完整错误信息
    error_stack TEXT,                        -- 解析出的堆栈
    screenshot_url TEXT,                     -- 截图（如有）
    
    -- AI分析结果
    root_cause TEXT,                         -- 根因分析
    affected_files TEXT[],                   -- 影响的文件列表
    fix_diff TEXT,                           -- 修复的代码Diff
    fix_risk VARCHAR(10),                    -- low/medium/high
    
    -- 执行状态
    status VARCHAR(20) DEFAULT 'pending',    -- pending/executed/verified/failed/ignored
    executed_by VARCHAR(20),               -- ai/manual
    verified_result TEXT,                    -- 验证结果（构建/测试输出）
    
    -- 关联
    similar_bug_id UUID REFERENCES bug_records(id),
    
    created_at TIMESTAMPTZ DEFAULT now(),
    updated_at TIMESTAMPTZ DEFAULT now()
);
